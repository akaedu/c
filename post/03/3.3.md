##3.3 scanf
###3.3.1 从键盘输入一个整数
    #include <stdio.h>

    void main()
    {
        int i;
        scanf("%d", &i);
        printf("%d\n", i);
        
        return 0;
    }
###3.3.2 从键盘输入一个字符串
    #include <stdio.h>

    void main()
    {
        char str[100];
        scanf("%s", str);
        printf("%s\n", str);

        return 0;
    }
###3.3.3 以空格为分格输入多个整数
    #include <stdio.h>

    void main()
    {
        int i, j, k;
        scanf("%d %d %d", &i, &j, &k);
        printf("i = %d, j = %d, k = %d\n", i, j, k);

        return 0;
    }
