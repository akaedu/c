##4.2 switch case

switch语句可以产生具有多个分支的控制流程。它的格式是：

	switch (控制表达式) {
	case 常量表达式： 语句列表
	case 常量表达式： 语句列表
	...
	default： 语句列表
	}

例如：

实现一个函数，他能根据参数进行加减乘除运算；

	#include <stdio.h>
	int main(void)
	{
		int a, b;
		char c;
		printf("please input(num1 + num2):");
		scanf("%d %c %d", &a, &c, &b);
		switch(c)
		{
			case '+': 
				printf("%d %c %d = %d\n", a, c, b, a + b);
				break;
			case '-': 
				printf("%d %c %d = %d\n", a, c, b, a - b);
				break;
			case '*': 
				printf("%d %c %d = %d\n", a, c, b, a * b);
				break;
			case '/': 
				printf("%d %c %d = %d\n", a, c, b, a / b);
				break;
			case '%': 
				printf("%d %c %d = %d\n", a, c, b, a % b);
				break;
			default: 
				printf("Invalid!\n");
				break;
		}
		return 0;
	}
